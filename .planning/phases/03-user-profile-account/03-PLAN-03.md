    File content:
    ```typescript
    import { client } from '@/api/client'
    import type { UserProfile } from '@/types'

    export interface UpdateProfilePayload {
      firstName?: string | null
      lastName?: string | null
      language?: string
      defaultPageSize?: string | null
      defaultInstrumentId?: string | null
    }

    export interface InstrumentOption {
      id: string
      name: string
    }

    export async function getMe(): Promise<UserProfile> {
      const { data } = await client.get<UserProfile>('/users/me')
      return data
    }

    export async function updateMe(payload: UpdateProfilePayload): Promise<UserProfile> {
      const { data } = await client.patch<UserProfile>('/users/me', payload)
      return data
    }

    export async function uploadAvatar(file: File): Promise<{ avatarUrl: string }> {
      const formData = new FormData()
      formData.append('file', file)
      const { data } = await client.post<{ avatarUrl: string }>('/users/me/avatar', formData, {
        headers: { 'Content-Type': 'multipart/form-data' },
      })
      return data
    }

    export async function requestDeletion(): Promise<UserProfile> {
      const { data } = await client.post<UserProfile>('/users/me/deletion')
      return data
    }

    export async function cancelDeletion(): Promise<UserProfile> {
      const { data } = await client.delete<UserProfile>('/users/me/deletion')
      return data
    }

    export async function getInstruments(): Promise<InstrumentOption[]> {
      const { data } = await client.get<InstrumentOption[]>('/instruments')
      return data
    }
    ```

    Implementation notes:
    - `UpdateProfilePayload` fields are all optional — PATCH is partial; caller omits unchanged fields
    - `uploadAvatar`: explicit `'Content-Type': 'multipart/form-data'` header lets Axios set the correct boundary; passing FormData to Axios without the header would also work in most browsers, but being explicit is safer
    - `requestDeletion` uses POST with no body (the backend schedules deletion for ~30 days from now)
    - `cancelDeletion` uses DELETE (no body)
    - `getInstruments` is at `/instruments` (not `/users/me/instruments`) — it is a metadata endpoint
    - All functions use `client` (authenticated) — NOT `rawClient`
    - `import type { UserProfile }` — verbatimModuleSyntax; UserProfile is a type-only import
    - No enums, no namespaces, no parameter properties — erasableSyntaxOnly
    - `UpdateProfilePayload` and `InstrumentOption` are exported — Profile page needs these types

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| ProfilePage → profileApi | All calls go through the authenticated client — token attached by interceptor |
| File upload | FormData with File object — size validation happens in ProfilePage before this function is called |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation |
|-----------|----------|-----------|-------------|------------|
| T-03-03-01 | Tampering | uploadAvatar payload | mitigate | Client validates file size ≤ 2 MB before calling uploadAvatar; MIME type enforced by file input accept attribute; backend performs its own validation on the file |
| T-03-03-02 | Information Disclosure | UpdateProfilePayload | accept | Only profile fields included; no sensitive data (no password, no token); sent over HTTPS via authenticated client |
| T-03-03-03 | Elevation of Privilege | getMe / updateMe using client | accept | client attaches the user's own access token; backend enforces /users/me to the authenticated user only |
</threat_model>
